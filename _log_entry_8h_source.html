230 LogEntry& LogEntry::obfuscatePrivateData(const char* key, const std::string& value) {
231  // if value contains any private label, obfuscate the section after the label
232  // since it can (but shouldn't) contain multiple, obfuscate from the earliest one found onward
233  auto firstPosition = value.length();
234 
235  for (auto privateLabel : getPrivateLabelDenyList()) {
236  auto it = std::search(
237  value.begin(),
238  value.end(),
239  privateLabel.begin(),
240  privateLabel.end(),
241  [](char valueChar, char denyListChar) { return std::tolower(valueChar) == std::tolower(denyListChar); });
242  if (it != value.end()) {
243  // capture the least value
244  auto thisPosition = std::distance(value.begin(), it) + privateLabel.length();
245  if (thisPosition < firstPosition) {
246  firstPosition = thisPosition;
247  }
248  }
249  }
250 
251  if (firstPosition <= value.length()) {
252  // hash everything after the label itself
253  auto labelPart = value.substr(0, firstPosition);
254  auto obfuscatedPart = std::to_string(std::hash<std::string>{}(value.substr(firstPosition)));
255  return d(key, labelPart + obfuscatedPart);
256  }
257  return d(key, value);
258 }
